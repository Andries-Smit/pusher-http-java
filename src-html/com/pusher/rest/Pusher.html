053        /**
054         * Construct an instance of the Pusher object through which you may interact with the Pusher API.
055         * <p>
056         * The parameters to use are found on your dashboard at https://app.pusher.com and are specific per App.
057         * <p>
058         * @param appId The ID of the App you will to interact with.
059         * @param key The App Key, the same key you give to websocket clients to identify your app when they connect to Pusher.
060         * @param secret The App Secret. Used to sign requests to the API, this should be treated as sensitive and not distributed.
061         */
062        public Pusher(final String appId, final String key, final String secret) {
063            Prerequisites.nonEmpty("appId", appId);
064            Prerequisites.nonEmpty("key", key);
065            Prerequisites.nonEmpty("secret", secret);
066            Prerequisites.isValidSha256Key("secret", secret);
067    
068            this.appId = appId;
069            this.key = key;
070            this.secret = secret;
071    
072            configure();
073        }
074    
075        public Pusher(final String url) {
076            Prerequisites.nonNull("url", url);
077    
078            final Matcher m = HEROKU_URL.matcher(url);
079            if (m.matches()) {
080                this.scheme = m.group(1);
081                this.key = m.group(2);
082                this.secret = m.group(3);
083                this.host = m.group(4);
084                this.appId = m.group(5);
085            }
086            else {
087                throw new IllegalArgumentException("URL '" + url + "' does not match pattern '<scheme>://<key>:<secret>@<host>[:<port>]/apps/<appId>'");
088            }
089    
090            Prerequisites.isValidSha256Key("secret", secret);
091            configure();
092        }

316        /**
317         * Make a generic REST call to the Pusher API.
318         * <p>
319         * The body should be a UTF-8 encoded String
320         * <p>
321         * See: http://pusher.com/docs/rest_api
322         * <p>
323         * NOTE: the path specified here is relative to that of your app. For example, to access
324         * the channel list for your app, simply pass "/channels". Do not include the "/apps/[appId]"
325         * at the beginning of the path.
326         */
327        public Result post(final String path, final String body) {
328            final String fullPath = "/apps/" + appId + path;
329            final URI uri = SignatureUtil.uri("POST", scheme, host, fullPath, body, key, secret, Collections.<String, String>emptyMap());
330    
331            final StringEntity bodyEntity = new StringEntity(body, "UTF-8");
332            bodyEntity.setContentType("application/json");
333    
334            final HttpPost request = new HttpPost(uri);
335            request.setEntity(bodyEntity);
336    
337            return httpCall(request);
338        }
339    
340        Result httpCall(final HttpRequestBase request) {
341            final RequestConfig config = RequestConfig.custom()
342                    .setSocketTimeout(requestTimeout)
343                    .setConnectionRequestTimeout(requestTimeout)
344                    .setConnectTimeout(requestTimeout)
345                    .build();
346            request.setConfig(config);
347    
348            try {
349                final HttpResponse response = client.execute(request);
350    
351                final ByteArrayOutputStream baos = new ByteArrayOutputStream();
352                response.getEntity().writeTo(baos);
353                final String responseBody = new String(baos.toByteArray(), "UTF-8");
354    
355                return Result.fromHttpCode(response.getStatusLine().getStatusCode(), responseBody);
356            }
357            catch (final IOException e) {
358                return Result.fromException(e);
359            }
360        }

384        /**
385         * Generate authentication response to authorise a user on a private channel
386         * <p>
387         * The return value is the complete body which should be returned to a client requesting authorisation.
388         */
389        public String authenticate(final String channel, final String socketId) {
390            Prerequisites.nonNull("socketId", socketId);
391            Prerequisites.nonNull("channel", channel);
392    
393            if (channel.startsWith("presence-")) {
394                throw new IllegalArgumentException("This method is for private channels, use authenticate(String, String, PresenceUser) to authenticate for a presence channel.");
395            }
396            if (!channel.startsWith("private-")) {
397                throw new IllegalArgumentException("Authentication is only applicable to private and presence channels");
398            }
399    
400            final String signature = SignatureUtil.sign(socketId + ":" + channel, secret);
401            return BODY_SERIALISER.toJson(new AuthData(key, signature));
402        }
403    
404        /**
405         * Generate authentication response to authorise a user on a private channel
406         * <p>
407         * The return value is the complete body which should be returned to a client requesting authorisation.
408         */
409        public String authenticate(final String channel, final String socketId, final PresenceUser user) {
410            Prerequisites.nonNull("socketId", socketId);
411            Prerequisites.nonNull("channel", channel);
412            Prerequisites.nonNull("user", user);
413    
414            if (channel.startsWith("private-")) {
415                throw new IllegalArgumentException("This method is for presence channels, use authenticate(String, String) to authenticate for a private channel.");
416            }
417            if (!channel.startsWith("presence-")) {
418                throw new IllegalArgumentException("Authentication is only applicable to private and presence channels");
419            }
420    
421            final String channelData = BODY_SERIALISER.toJson(user);
422            final String signature = SignatureUtil.sign(socketId + ":" + channel + ":" + channelData, secret);
423            return BODY_SERIALISER.toJson(new AuthData(key, signature, channelData));
424        }

430        public Validity validateWebhookSignature(final String xPusherKeyHeader, final String xPusherSignatureHeader, final String body) {
431            if (!xPusherKeyHeader.trim().equals(key)) {
432                // We can't validate the signature, because it was signed with a different key to the one we were initialised with.
433                return Validity.SIGNED_WITH_WRONG_KEY;
434            }
435    
436            final String recalculatedSignature = SignatureUtil.sign(body, secret);
437            return xPusherSignatureHeader.trim().equals(recalculatedSignature) ? Validity.VALID : Validity.INVALID;
438        }
