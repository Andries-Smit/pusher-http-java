001    package com.pusher.rest;
002    
003    import java.io.UnsupportedEncodingException;
004    import java.net.URI;
005    import java.net.URISyntaxException;
006    import java.security.InvalidKeyException;
007    import java.security.MessageDigest;
008    import java.security.NoSuchAlgorithmException;
009    import java.util.Arrays;
010    import java.util.HashMap;
011    import java.util.Map;
012    import java.util.Map.Entry;
013    
014    import javax.crypto.Mac;
015    import javax.crypto.spec.SecretKeySpec;
016    
017    import org.apache.commons.codec.binary.Hex;
018    import org.apache.http.client.utils.URIBuilder;
019    
020    public class SignedRequest {
021    
022        public static URI uri(final String method,
023                              final String scheme,
024                              final String host,
025                              final String path,
026                              final String body,
027                              final String key,
028                              final String secret,
029                              final Map<String, String> extraParams) {
030            try {
031                final Map<String, String> allParams = new HashMap<String, String>(extraParams);
032                allParams.put("auth_key", key);
033                allParams.put("auth_version", "1.0");
034                allParams.put("auth_timestamp", new Long(System.currentTimeMillis() / 1000).toString());
035                allParams.put("body_md5", bodyMd5(body));
036    
037                // This is where the auth gets a bit weird. The query params for the request must include
038                // the auth signature which is a signature over all the params except itself.
039                allParams.put("auth_signature", sign(buildSignatureString(method, path, allParams), secret));
040    
041                final URIBuilder b = new URIBuilder()
042                        .setScheme(scheme)
043                        .setHost(host)
044                        .setPath(path);
045    
046                for (final Entry<String, String> e : allParams.entrySet()) {
047                    b.setParameter(e.getKey(), e.getValue());
048                }
049    
050                return b.build();
051            }
052            catch (final URISyntaxException e) {
053                throw new RuntimeException("Could not build URI", e);
054            }
055            catch (final InvalidKeyException e) {
056                /// We validate this when the key is first provided, so we should never encounter it here.
057                throw new RuntimeException("Invalid secret key", e);
058            }
059            // If either of these doesn't exist, we're pretty much out of luck.
060            catch (final NoSuchAlgorithmException e) {
061                throw new RuntimeException("The Pusher REST client requires MD5 and HmacSHA256 support", e);
062            }
063            catch (final UnsupportedEncodingException e) {
064                throw new RuntimeException("The Pusher REST client needs UTF-8 support", e);
065            }
066        }
067    
068        private static String bodyMd5(final String body) throws NoSuchAlgorithmException, UnsupportedEncodingException {
069            final MessageDigest md = MessageDigest.getInstance("MD5");
070            final byte[] digest = md.digest(body.getBytes("UTF-8"));
071            return Hex.encodeHexString(digest);
072        }
073    
074        private static String sign(final String input, final String secret) throws NoSuchAlgorithmException, InvalidKeyException, IllegalStateException, UnsupportedEncodingException {
075            final Mac mac = Mac.getInstance("HmacSHA256");
076            mac.init(new SecretKeySpec(secret.getBytes(), "SHA256"));
077    
078            final byte[] digest = mac.doFinal(input.getBytes("UTF-8"));
079            return Hex.encodeHexString(digest);
080        }
081    
082        // Visible for testing
083        static String buildSignatureString(final String method, final String path, final Map<String, String> queryParams) {
084            final StringBuilder sb = new StringBuilder();
085            sb.append(method)
086                .append('\n')
087                .append(path)
088                .append('\n');
089    
090            final String[] keys = queryParams.keySet().toArray(new String[0]);
091            Arrays.sort(keys);
092    
093            boolean first = true;
094            for (final String key : keys) {
095                if (!first) sb.append('&');
096                else first = false;
097    
098                sb.append(key)
099                    .append('=')
100                    .append(queryParams.get(key));
101            }
102    
103            return sb.toString();
104        }
105    }
