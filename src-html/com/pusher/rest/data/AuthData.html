001    package com.pusher.rest.data;
002    
003    public class AuthData {
004    
005        private final String auth;
006        private final String channelData;
007    
008        /**
009         * Private channel constructor
010         */
011        public AuthData(final String key, final String signature) {
012            this(key, signature, null);
013        }
014    
015        /**
016         * Presence channel constructor
017         */
018        public AuthData(final String key, final String signature, final String channelData) {
019            this.auth = key + ":" + signature;
020            this.channelData = channelData;
021        }
022    
023        public String getAuth() {
024            return auth;
025        }
026    
027        public String getChannelData() {
028            return channelData;
029        }
030    }
