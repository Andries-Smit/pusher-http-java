001    package com.pusher.rest.data;
002    
003    import com.pusher.rest.util.Prerequisites;
004    
005    /**
006     * Represents a precence channel "user", that is a user from the domain of your application.
007     */
008    public class PresenceUser {
009    
010        private final Object userId;
011        private final Object userInfo;
012    
013        /**
014         * Represents a presence channel user with no additional data associated other than the userId
015         */
016        public PresenceUser(final String userId) {
017            this((Object)userId, null);
018        }
019    
020        /**
021         * Represents a presence channel user with no additional data associated other than the userId
022         */
023        public PresenceUser(final Number userId) {
024            this((Object)userId, null);
025        }
026    
027        /**
028         * Represents a presence channel user and a map of data associated with the user
029         */
030        public PresenceUser(final String userId, final Object userInfo) {
031            this((Object)userId, userInfo);
032        }
033    
034        /**
035         * Represents a presence channel user and a map of data associated with the user
036         */
037        public PresenceUser(final Number userId, final Object userInfo) {
038            this((Object)userId, userInfo);
039        }
040    
041        /**
042         * There's not really a great way to accept either a string or numeric value in a typesafe way,
043         * so this will have to do.
044         */
045        private PresenceUser(final Object userId, final Object userInfo) {
046            Prerequisites.nonNull("userId", userId);
047    
048            this.userId = userId;
049            this.userInfo = userInfo;
050        }
051    
052        public Object getUserId() {
053            return userId;
054        }
055    
056        public Object getUserInfo() {
057            return userInfo;
058        }
059    }
