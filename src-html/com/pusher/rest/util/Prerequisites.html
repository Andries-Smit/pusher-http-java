001    package com.pusher.rest.util;
002    
003    import java.security.InvalidKeyException;
004    import java.security.NoSuchAlgorithmException;
005    import java.util.Arrays;
006    import java.util.HashSet;
007    import java.util.List;
008    import java.util.Map;
009    import java.util.Set;
010    
011    import javax.crypto.Mac;
012    import javax.crypto.spec.SecretKeySpec;
013    
014    public final class Prerequisites {
015    
016        private static final Set<String> RESERVED_QUERY_KEYS = new HashSet<String>(
017                Arrays.asList(new String[] { "auth_key", "auth_timestamp", "auth_version", "auth_signature", "body_md5" }));
018    
019        public static void nonNull(final String name, final Object ref) {
020            if (ref == null) throw new IllegalArgumentException("Parameter [" + name + "] must not be null");
021        }
022    
023        public static void nonEmpty(final String name, final String ref) {
024            nonNull(name, ref);
025            if (ref.length() == 0) throw new IllegalArgumentException("Parameter [" + name + "] must not be empty");
026        }
027    
028        public static void maxLength(final String name, final int max, final List<?> ref) {
029            if (ref.size() > max) throw new IllegalArgumentException("Parameter [" + name + "] must have size < " + max);
030        }
031    
032        public static void noNullMembers(final String name, final List<?> ref) {
033            for (Object e : ref) {
034                if (e == null) throw new IllegalArgumentException("Parameter [" + name + "] must not contain null elements");
035            }
036        }
037    
038        public static void noReservedKeys(final Map<String, String> params) {
039            for (String k : params.keySet()) {
040                if (RESERVED_QUERY_KEYS.contains(k.toLowerCase())) {
041                    throw new IllegalArgumentException("Query parameter key [" + k + "] is reserved and should not be submitted. It will be generated by the signature generation.");
042                }
043            }
044        }
045    
046        public static void isValidSha256Key(final String name, final String key) {
047            try {
048                Mac mac = Mac.getInstance("HmacSHA256");
049                mac.init(new SecretKeySpec(key.getBytes(), "SHA256"));
050                // If that goes OK, then we're good to go
051            }
052            catch (final NoSuchAlgorithmException e) {
053                // Out of luck.
054                throw new RuntimeException("The Pusher REST client requires HmacSHA256 support", e);
055            }
056            catch (final InvalidKeyException e) {
057                // Failed the test
058                throw new IllegalArgumentException("Parameter [" + name + "] must be a valid SHA256 key", e);
059            }
060        }
061    }
