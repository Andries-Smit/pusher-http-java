001    package com.pusher.rest.util;
002    
003    import java.security.InvalidKeyException;
004    import java.security.NoSuchAlgorithmException;
005    import java.util.List;
006    
007    import javax.crypto.Mac;
008    import javax.crypto.spec.SecretKeySpec;
009    
010    public final class Prerequisites {
011    
012        public static void nonNull(final String name, final Object ref) {
013            if (ref == null) throw new IllegalArgumentException("Parameter [" + name + "] must not be null");
014        }
015    
016        public static void nonEmpty(final String name, final String ref) {
017            nonNull(name, ref);
018            if (ref.length() == 0) throw new IllegalArgumentException("Parameter [" + name + "] must not be empty");
019        }
020    
021        public static void maxLength(final String name, final int max, final List<?> ref) {
022            if (ref.size() > max) throw new IllegalArgumentException("Parameter [" + name + "] must have size < " + max);
023        }
024    
025        public static void noNullMembers(final String name, final List<?> ref) {
026            for (Object e : ref) {
027                if (e == null) throw new IllegalArgumentException("Parameter [" + name + "] must not contain null elements");
028            }
029        }
030    
031        public static void isValidSha256Key(final String name, final String key) {
032            try {
033                Mac mac = Mac.getInstance("HmacSHA256");
034                mac.init(new SecretKeySpec(key.getBytes(), "SHA256"));
035                // If that goes OK, then we're good to go
036            }
037            catch (final NoSuchAlgorithmException e) {
038                // Out of luck.
039                throw new RuntimeException("The Pusher REST client requires HmacSHA256 support", e);
040            }
041            catch (final InvalidKeyException e) {
042                // Failed the test
043                throw new IllegalArgumentException("Parameter [" + name + "] must be a valid SHA256 key", e);
044            }
045        }
046    }
